class Veiculo {
    protected:
        string marca;
        int ano;
    public:
        Veiculo(string m, int a) : marca(m), ano(a) {}
};

class ModeloCarro : public Veiculo {
    private:
        string modelo;
    public:
        ModeloCarro(string m, int a, string model) : Veiculo(m, a), modelo(model) {}
       
};
